# dotfiles

I usually keep the repo in `~/.dotfiles`:

```shell
git clone https://github.com/pablopunk/dotfiles ~/.dotfiles
```

You can install all dotfiles at once:

```shell
$ bash install.sh
'/home/pablopunk/.bash_aliases' -> '/home/pablopunk/.dotfiles/bash/bash_aliases'
'/home/pablopunk/.bash_functions' -> '/home/pablopunk/.dotfiles/bash/bash_functions'
'/home/pablopunk/.bash_hello' -> '/home/pablopunk/.dotfiles/bash/bash_hello'
'/home/pablopunk/.bash_profile' -> '/home/pablopunk/.dotfiles/bash/bash_profile'
'/home/pablopunk/.bashrc' -> '/home/pablopunk/.dotfiles/bash/bashrc'
'/home/pablopunk/.inputrc' -> '/home/pablopunk/.dotfiles/bash/inputrc'
'/home/pablopunk/.config/deepin/deepin-terminal/config.conf' -> '/home/pablopunk/.dotfiles/deepin/config/deepin/deepin-terminal/config.conf'
'/home/pablopunk/.editorconfig' -> '/home/pablopunk/.dotfiles/editorconfig/editorconfig'
'/home/pablopunk/.hyper.js' -> '/home/pablopunk/.dotfiles/hyper/hyper.js'
'/home/pablopunk/.config/nvim/init.vim' -> '/home/pablopunk/.dotfiles/nvim/config/nvim/init.vim'
'/home/pablopunk/.tern-project' -> '/home/pablopunk/.dotfiles/tern/tern-project'
'/home/pablopunk/.tmux.conf' -> '/home/pablopunk/.dotfiles/tmux/tmux.conf'
'/home/pablopunk/.tmuxinator' -> '/home/pablopunk/.dotfiles/tmuxinator'
'/home/pablopunk/.vimrc' -> '/home/pablopunk/.dotfiles/vim/vimrc'
```
